window.onload = function() { var urlParams = new URLSearchParams(window.location.search); var iframeSrc = urlParams.get("id"); if (iframeSrc) { var iframe = document.createElement("iframe"); iframe.src = iframeSrc; iframe.frameBorder = "0"; iframe.allowFullscreen = true; iframe.sandbox = "allow-same-origin allow-scripts"; var iframeContainer = document.createElement("div"); iframeContainer.className = "ViostreamIframe"; iframeContainer.appendChild(iframe); document.body.appendChild(iframeContainer); } };
